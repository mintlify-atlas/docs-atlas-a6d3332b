## Overview

The Creative! frontend is built with **Vue 3.5+** using the Composition API and TypeScript. State management uses a combination of **provide/inject** for hierarchical data and **exported refs** from composables for shared global state.

## Project Structure

```

├── App.vue                    # Root component with state providers
├── components/
│   ├── Modes.vue              # Replay/Analysis mode switcher
```

## Root Component (App.vue)

### State Initialization

The root component sets up all global state and provides it to child components:

```vue
<script setup lang="ts">
import { ref, provide, Ref, useTemplateRef } from "vue"
import { useVideo } from "./components/store/hooks/useVideo"
import useBreakpoint from "./components/store/hooks/useBreakpoint"
import { videoUrl, analysisData } from "./components/store/hooks/useFileUpload"

// Mode state (replay vs analysis)
const mode: Ref<"replay" | "analysis"> = ref("replay")
const editing: SelectedTool = ref(null)

// Video playback controls
const videoElement = useTemplateRef("video-playback")
const playbackControls = useVideo(videoElement)

// Breakpoint/analysis state
const breakpointStore = useBreakpoint()

// Provide to all descendants
provide("mode", { mode, toggleMode })
provide("video", { videoElement, playbackControls })
provide("breakpointStore", breakpointStore)
provide("editing", editing)
provide("activeColor", ref("red"))
</script>
```

### useVideo() - Playback Control

<Info>
  Manages all video playback state including play/pause, seeking, variable-speed playback, and trim boundaries.
</Info>

#### Core Features

  <Tab title="Implementation">
    ```typescript
    // src/components/store/hooks/useVideo.ts
    export function useVideo(initialSrc: VideoRef = ref(null)): VideoHook {
      const _videoElementRef = initialSrc
      const _currentFrame = ref<number | null>(null)
      const _playbackDirection = ref<boolean | null>(null)
      const _playbackSpeed = ref<PlaybackSpeed>(1)
      const _customVideoEnd = ref<number | null>(null)
      const _customVideoStart = ref<number | null>(null)
      
      // Watch for playback changes
      watch([_playbackDirection, _playbackSpeed], ([newDir, newSpeed]) => {
        if (!_videoElementRef.value) return
        
        _clearAllIntervals()
        _cancelLoop()
        
        switch (newDir) {
          case true:  // Forward
            if (newSpeed === 1) {
              _videoElementRef.value.playbackRate = 1
              _videoElementRef.value.play()
              _loop()  // Start progress tracking
            } else {
              _playManualSeek(newSpeed, true)
            }
            break
          case false: // Rewind
            _playManualSeek(newSpeed, false)
            break
          case null:  // Paused
            _videoElementRef.value.pause()
            break
        }
      })
      
      return { /* ... public API ... */ }
    }
    ```
    **Reference:** `src/components/store/hooks/useVideo.ts:9-396`
  </Tab>
  <Tab title="Manual Seeking">
    For speeds >1x or any rewind, native playback is insufficient. The composable uses interval-based seeking:
    
    ```typescript
    const _playManualSeek = (rate: PlaybackSpeed, direction: boolean) => {
      const TARGET_FPS = 30
      const INTERVAL_MS = 1000 / TARGET_FPS
      
      _intervalTracker = setInterval(() => {
        const video = _videoElementRef.value
        if (!video) return
        
        const jumpSeconds = (rate * INTERVAL_MS) / 1000
        
        if (!direction) {  // Rewind
          if (video.currentTime <= _startTime.value) {
            pause()
            return
          }
          video.currentTime -= jumpSeconds
        } else {  // Fast forward
          const endPoint = _customVideoEnd.value || video.duration
          if (video.currentTime + jumpSeconds >= endPoint) {
            pause()
            return
          }
          video.currentTime += jumpSeconds
        }
        
        _currentFrame.value = video.currentTime
      }, INTERVAL_MS)
    }
    ```
    **Reference:** `src/components/store/hooks/useVideo.ts:260-321`
  </Tab>
</Tabs>

#### Trim Boundaries

```typescript
const setTrim = (start: number | null, end: number | null) => {
  if (start === null || end === null) {
    _customVideoStart.value = null
    _customVideoEnd.value = null
    return
  }
  
  // start and end are relative to current visible clip
  const currentAbsoluteStart = _startTime.value || 0
  const newStart = currentAbsoluteStart + start
  const newEnd = currentAbsoluteStart + end
  
  _customVideoStart.value = newStart
  _customVideoEnd.value = newEnd
  
  // Jump to trim start if current position is outside bounds
  if (_videoElementRef.value) {
    const currentTime = _videoElementRef.value.currentTime
    if (currentTime < newStart || currentTime > newEnd) {
      _videoElementRef.value.currentTime = newStart
      _currentFrame.value = newStart
    }
  }
}
```

**Reference:** `src/components/store/hooks/useVideo.ts:346-371`

<Note>
  When exporting analysis files, trim boundaries are saved to `data.json` as `videoStart` and `videoEnd`. On import, `setTrim()` is called in `App.vue:44-46` to restore the trimmed segment.
</Note>

### useBreakpoint() - Analysis Data

<Info>

#### Data Structure

```typescript

  invertedY: boolean

  fileBlob: Blob
  filename?: string  // Set during export

}
```

#### ID Generation

<Warning>
  IDs for content items are auto-incremented within the composable. When importing data, `loadBreakpointData()` scans all existing IDs and sets `currentId` to `maxId + 1` to prevent collisions.
</Warning>

```typescript
const loadBreakpointData = (breakpoints: Breakpoint[]) => {
  _state.breakpoints = breakpoints
  
  // Update currentId to avoid collisions
  let maxId = 0
  breakpoints.forEach((bp) => {
    bp.textContent?.forEach((tc) => (maxId = Math.max(maxId, tc.id)))
    bp.voiceContent?.forEach((vc) => (maxId = Math.max(maxId, vc.id)))
    if (bp.drawingContent) maxId = Math.max(maxId, bp.drawingContent.id)
  })
  currentId = maxId + 1
}
```

## Component Communication Patterns

### Provide/Inject (Hierarchical)

Used for state that flows down the component tree:

```vue
<!-- Parent: App.vue -->
<script setup>
const mode = ref("replay")
provide("mode", { mode, toggleMode })
</script>

<!-- Child: Modes.vue -->
<script setup>
const { mode, toggleMode } = inject("mode")
</script>
<button @click="toggleMode">
  {{ mode === 'replay' ? 'Analysis' : 'Replay' }}
</button>
```

### Shared Refs (Global)

Used for state that needs to be accessed from multiple unrelated components:

```typescript
// useFileUpload.ts (export)
export const videoUrl = ref<string | null>(null)

// App.vue (import and use)
import { videoUrl } from "./store/hooks/useFileUpload"
<video v-if="videoUrl" :src="videoUrl" />

// Any other component (import and use)
import { videoUrl } from "@/store/hooks/useFileUpload"
const hasVideo = computed(() => videoUrl.value !== null)
```

## Performance Considerations

### Video Loading

<Note>
  Always unmount the video element before changing `videoUrl.value` to ensure the `@loadedmetadata` event fires:
  
  ```typescript
  videoUrl.value = null  // Force unmount
  await nextTick()
  videoUrl.value = newUrl  // Remount with new source
  ```
</Note>

### Large Archive Handling

For `.an` files, unpacking is delegated to Rust to avoid JavaScript memory limits:

```typescript
// This bypasses JS engine completely
const dataJson = await invoke<string>("unpack_analysis_file", {
  archivePath: selected,
  targetDir: extractionTarget
})
// Only the parsed JSON is loaded into JS memory
```

**Reference:** `src/components/store/hooks/useFileUpload.ts:75-78`

### Audio Blob Reconstruction

Audio files are fetched individually after unpacking (lazy loading):

```typescript
for (const vc of bp.voiceContent) {
  if (vc.filename) {
    const audioUrl = buildAssetUrl(port, audioPath)
    const res = await fetch(audioUrl)
    vc.fileBlob = await res.blob()  // Load into memory
  }
}
```

This approach:
- Reduces initial load time
- Only loads audio that exists in breakpoints
- Uses streaming fetch (handled by Tauri/browser)
